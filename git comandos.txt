# inicia o git(repositorio) no seu projeto
git init
# verificar alterações de pastas e arquivos no projeto
git status

# adicionar todos os arquivos e pastas modificados, ao stage alterações
git add .

# cria e descreve um ponto na historia
git commit -m "message here"

# historico de commits do projeto
git log3

# remover arquivo que foi upado no git add
git rm --cached -f <arquivo>

#restaurar o arquivo
git restore <arquivo>

#recuperar um arquivo deletado
 uso o checkout para ir no commit que tem o arquivo antes de ser deletado
 e uso checkout <id> -- <nome-arquivo>


# enviar alteracao para o github
git push

#puxa modificaçoes do repositorio remoto para o repositorio local
git pull
